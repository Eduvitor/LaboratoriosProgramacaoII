CREATE TABLE IF NOT EXISTS public.cursos
(
    id integer NOT NULL,
    nome_curso character varying(30) COLLATE pg_catalog."default" NOT NULL,
    descricao text COLLATE pg_catalog."default",
    turno character varying(10) COLLATE pg_catalog."default",
    coordenadores character varying(50) COLLATE pg_catalog."default",
    semestre character varying(20) COLLATE pg_catalog."default",
    CONSTRAINT cursos_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.cursos
    OWNER to postgres;
